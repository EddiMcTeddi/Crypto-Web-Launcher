'pos' : 'neg', arr2 = chg >= 0 ? '▲' : '▼'; return ` ${c.sym} ${fmtUsd(c.data.usd)} ${arr2} ${Math.abs(chg).toFixed(1)}% `; }).join(''); } document.getElementById('mini-crypto-content').innerHTML = `
${makeRows(leftCoins)}
${makeRows(rightCoins)}
`; // Ticker (alle 8) const items = coins.map(c => { const chg = c.data.usd_24h_change, cls = chg >= 0 ? 'pos' : 'neg', arr2 = chg >= 0 ? '▲' : '▼'; return `${c.sym}${fmtUsd(c.data.usd)}${arr2} ${Math.abs(chg).toFixed(2)}%`; }); document.getElementById('ticker-track').innerHTML = [...items,...items,...items,...items].join(''); } catch(e) { document.getElementById('mini-crypto-content').innerHTML = 'FEIL VED HENTING'; } }
